    <script>
        // Simple Countdown Timer
        function startTimer(duration, displayH, displayM, displayS) {
            var timer = duration, hours, minutes, seconds;
            setInterval(function () {
                hours = parseInt(timer / 3600, 10);
                minutes = parseInt((timer % 3600) / 60, 10);
                seconds = parseInt(timer % 60, 10);

                displayH.textContent = hours < 10 ? "0" + hours : hours;
                displayM.textContent = minutes < 10 ? "0" + minutes : minutes;
                displayS.textContent = seconds < 10 ? "0" + seconds : seconds;

                if (--timer < 0) { timer = duration; }
            }, 1000);
        }

        window.onload = function () {
            var threeHours = 60 * 60 * 3;
            startTimer(threeHours, document.querySelector('#hours'), document.querySelector('#mins'), document.querySelector('#secs'));
        };
    </script>
